# Timer

This is the `RusticUI` Timer Component. It is fairly simple. You just wrap whatever content you want inside of `<Timer>` and `</Timer>`.

import { Canvas, Meta, Story, ArgsTable } from "@storybook/addon-docs";
import { Timer } from "./Timer";

<Meta title="Timer" component={Timer} />

export const Template = (args) => <Timer {...args}>{args.children}</Timer>;

<Canvas>
  <Story
    name="Timer"
    args={{
      children: <h1>Example Timer</h1>,
    }}
  >
    {Template.bind({})}
  </Story>
</Canvas>
;
